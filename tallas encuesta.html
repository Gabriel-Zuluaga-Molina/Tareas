# formula: n / Σ(1/xi)

Media de orden h (Media Potencia)

# Fórmula general: [ (1/n) * Σ(xi^h) ]^(1/h)
# h = 1 → media aritmética
# h = -1 → media armónica
# h → 0 → media geométrica

calcularmediah = function(x, h){
  if(h == 0){
    return(geoMean(x))
  } else{
    return((mean(x^h)^(1/h)))
  }
}

Momentos Estadísticos

# Los momentos describen la forma de la distribución
# Momento central de orden k: (1/n) * Σ(xi - media)^k

calcularmoce = function(x, k) {
  mediax = mean(x)
  return(mean((x - mediax)^k))
}

cat("Momento central de orden 1 (siempre ~0):",calcularmoce(encuesta, 1), "\n")
